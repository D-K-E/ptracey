19 struct quat_product<Architecture::SSE, Derived, OtherDerived, float>
20 {
21  enum {
22  AAlignment = traits<Derived>::Alignment,
23  BAlignment = traits<OtherDerived>::Alignment,
24  ResAlignment = traits<Quaternion<float> >::Alignment
25  };
26  static inline Quaternion<float> run(const QuaternionBase<Derived>& _a, const QuaternionBase<OtherDerived>& _b)
27  {
28  evaluator<typename Derived::Coefficients> ae(_a.coeffs());
29  evaluator<typename OtherDerived::Coefficients> be(_b.coeffs());
30  Quaternion<float> res;
31  const Packet4f mask = _mm_setr_ps(0.f,0.f,0.f,-0.f);
32  Packet4f a = ae.template packet<AAlignment,Packet4f>(0);
33  Packet4f b = be.template packet<BAlignment,Packet4f>(0);
34  Packet4f s1 = pmul(vec4f_swizzle1(a,1,2,0,2),vec4f_swizzle1(b,2,0,1,2));
35  Packet4f s2 = pmul(vec4f_swizzle1(a,3,3,3,1),vec4f_swizzle1(b,0,1,2,1));
36  pstoret<float,Packet4f,ResAlignment>(
37  &res.x(),
38  padd(psub(pmul(a,vec4f_swizzle1(b,3,3,3,3)),
39  pmul(vec4f_swizzle1(a,2,0,1,0),
40  vec4f_swizzle1(b,1,2,0,0))),
41  pxor(mask,padd(s1,s2))));
42 
43  return res;
44  }
45 };

88 template<class Derived, class OtherDerived>
89 struct quat_product<Architecture::SSE, Derived, OtherDerived, double>
90 {
91  enum {
92  BAlignment = traits<OtherDerived>::Alignment,
93  ResAlignment = traits<Quaternion<double> >::Alignment
94  };
95 
96  static inline Quaternion<double> run(const QuaternionBase<Derived>& _a, const QuaternionBase<OtherDerived>& _b)
97  {
98  const Packet2d mask = _mm_castsi128_pd(_mm_set_epi32(0x0,0x0,0x80000000,0x0));
99 
100  Quaternion<double> res;
101 
102  evaluator<typename Derived::Coefficients> ae(_a.coeffs());
103  evaluator<typename OtherDerived::Coefficients> be(_b.coeffs());
104 
105  const double* a = _a.coeffs().data();
106  Packet2d b_xy = be.template packet<BAlignment,Packet2d>(0);
107  Packet2d b_zw = be.template packet<BAlignment,Packet2d>(2);
108  Packet2d a_xx = pset1<Packet2d>(a[0]);
109  Packet2d a_yy = pset1<Packet2d>(a[1]);
110  Packet2d a_zz = pset1<Packet2d>(a[2]);
111  Packet2d a_ww = pset1<Packet2d>(a[3]);
112 
113  // two temporaries:
114  Packet2d t1, t2;
115 
116  /*
117  * t1 = ww*xy + yy*zw
118  * t2 = zz*xy - xx*zw
119  * res.xy = t1 +/- swap(t2)
120  */
121  t1 = padd(pmul(a_ww, b_xy), pmul(a_yy, b_zw));
122  t2 = psub(pmul(a_zz, b_xy), pmul(a_xx, b_zw));
123 #ifdef EIGEN_VECTORIZE_SSE3
124  EIGEN_UNUSED_VARIABLE(mask)
125  pstoret<double,Packet2d,ResAlignment>(&res.x(), _mm_addsub_pd(t1, preverse(t2)));
126 #else
127  pstoret<double,Packet2d,ResAlignment>(&res.x(), padd(t1, pxor(mask,preverse(t2))));
128 #endif
129 
130  /*
131  * t1 = ww*zw - yy*xy
132  * t2 = zz*zw + xx*xy
133  * res.zw = t1 -/+ swap(t2) = swap( swap(t1) +/- t2)
134  */
135  t1 = psub(pmul(a_ww, b_zw), pmul(a_yy, b_xy));
136  t2 = padd(pmul(a_zz, b_zw), pmul(a_xx, b_xy));
137 #ifdef EIGEN_VECTORIZE_SSE3
138  EIGEN_UNUSED_VARIABLE(mask)
139  pstoret<double,Packet2d,ResAlignment>(&res.z(), preverse(_mm_addsub_pd(preverse(t1), t2)));
140 #else
141  pstoret<double,Packet2d,ResAlignment>(&res.z(), psub(t1, pxor(mask,preverse(t2))));
142 #endif
143 
144  return res;
145 }
146 };
