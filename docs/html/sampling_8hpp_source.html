336 Real mis_estimate_f(
337  const VectorXf &alpha,
338  const std::vector<Real> &sample_set,
339  const std::function<Real(Real)> &f, const FnVec &p_N,
340  const std::map<int, int> &nb_samples_taken_from_p_N,
341  const std::vector<std::vector<Real>> &X_samples_by_pN,
342  int N) {
343  // \sum_{i=1}^N \alpha_i
344  Real alpha_sum = alpha.sum();
345 
346  // \frac{1}{M}
347  int M = 0;
348  for (int i = 1; i < N; i++) {
349  int nb_samples_taken_from_p_i =
350  nb_samples_taken_from_p_N.at(i);
351  M += nb_samples_taken_from_p_i;
352  }
353  Real frac_1_M = 1 / static_cast<Real>(M);
354 
355  auto p_c = [M, N, nb_samples_taken_from_p_N,
356  p_N](Real x) {
357  Real sumval = 0.0;
358  for (int i = 0; i < N; i++) {
359  int n_i = nb_samples_taken_from_p_N.at(i);
360  Real n_i_over_M =
361  static_cast<Real>(n_i) / static_cast<Real>(M);
362  auto p_i = p_N[i];
363  Real p_i_x = p_i(x);
364  sumval += n_i_over_M * p_i_x;
365  }
366  };

368  auto sum_alpha_pk = [p_N, alpha, N](Real x) -> Real {
369  Real sumval = 0;
370  for (int k = 0; k < N; k++) {
371  Real alpha_k = alpha[k];
372  auto p_k = p_N[k];
373  sumval += alpha_k * p_k(x);
374  }
375  return sumval;
376  };
377 
378  //\sum_{i=1}^N \sum_{j=1}^{n_i}
379  Real i_sum = 0.0;
380  for (int i = 0; i < N; i++) {
381  int n_i = nb_samples_taken_from_p_N.at(i);
382  Real j_sum = 0.0;
383  for (int j = 0; j < N; j++) {
384  // \frac{f(X_{ij})}{p_c{X_{ij}}}
385  Real X_ij = X_samples_by_pN[i][j];
386  Real f_x = f(X_ij);
387  Real p_cx = p_c(X_ij);
388  Real f_x_over_p_c = f_x / p_cx;
389 
392  Real alpha_k_sum = sum_alpha_pk(X_ij);
393  Real alpha_k_sum_over_pc = alpha_k_sum / p_c;
394  j_sum += f_x_over_p_c - alpha_k_sum_over_pc;
395  }
396  i_sum += j_sum;
397  }
398  Real m_i_sum = frac_1_M * i_sum;
399  return m_i_sum + alpha_sum;
