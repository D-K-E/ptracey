111  spd<Real> from_rgb(const vec3 &rgb) {
112  // from
113  // http://scottburns.us/fast-rgb-to-spectrum-conversion-for-reflectances/
114  // convert sRGB to linear rgb in range [0,1]
115  Real rgbm = rgb.max();
116  Real ins, ine;
117  if (rgbm > 1.0) {
118  ins = 0.0;
119  ine = 256.0;
120  } else {
121  ins = 0.0;
122  ine = 1.0;
123  }
124  auto lr = interp<Real>(rgb.r(), ins, ine, 0.0, 1.0);
125  auto lg = interp<Real>(rgb.g(), ins, ine, 0.0, 1.0);
126  auto lb = interp<Real>(rgb.b(), ins, ine, 0.0, 1.0);
127  //
128  auto rho_r_wave = rho_r.power();
129  auto rho_g_wave = rho_g.power();
130  auto rho_b_wave = rho_b.power();
131  auto rho = lr * rho_r_wave;
132  rho += lg * rho_g_wave;
133  rho += lb * rho_b_wave;
134  return spd<Real>(rho, rho_r.wavelengths());
135  }
136  spd<Real> from_rgb(const color &rgb) {
137  return from_rgb(vec3(rgb.r(), rgb.g(), rgb.b()));
138  }
