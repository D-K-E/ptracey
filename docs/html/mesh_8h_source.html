186  return true;
187  }

190  return false;
191  }
192 
193  if (nullptr == mIndices) {
194  return false;
195  }
196 
197  for (unsigned int i = 0; i < this->mNumIndices; ++i) {
198  if (mIndices[i] != o.mIndices[i]) {
199  return false;
200  }
201  }
202 
203  return true;
204  }
205
