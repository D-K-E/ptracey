82 bool sphere::hit(const ray &r, double t_min, double t_max,
83  hit_record &rec) const {
84  vec3 oc = r.origin() - center;
85  auto a = r.direction().length_squared();
86  auto half_b = dot(oc, r.direction());
87  auto c = oc.length_squared() - radius * radius;
88 
89  auto discriminant = half_b * half_b - a * c;
90  if (discriminant < 0)
91  return false;
92  auto sqrtd = sqrt(discriminant);
93 
94  // Find the nearest root that lies in the acceptable
95  // range.
96  auto root = (-half_b - sqrtd) / a;
97  if (root < t_min || t_max < root) {
98  root = (-half_b + sqrtd) / a;
99  if (root < t_min || t_max < root)
100  return false;
101  }
102 
103  rec.t = root;
104  rec.p = r.at(rec.t);
105  vec3 outward_normal = (rec.p - center) / radius;
106  rec.set_face_normal(r, outward_normal);
107  get_sphere_uv(outward_normal, rec.u, rec.v);
108  rec.mat_ptr = mat_ptr;
