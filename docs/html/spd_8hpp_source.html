521 Power get_cie_val(const spd &qlambda, const spd &cie_bar) {
522  Power sum = 0.0;
523  auto qwaves = qlambda.wavelengths();
524  auto qmaxwave = qlambda.max_wave();
525  auto qminwave = qlambda.min_wave();
526  auto qwsize = (uint)qwaves.size();
527  auto cie = cie_bar.resample_c(qminwave, qmaxwave, qwsize);
528  for (const auto &wave : qwaves) {
529  sum += qlambda[wave] * cie[wave];
530  }
531  auto scale = (qmaxwave - qminwave) / Real(qwsize);
532  return scale * sum;
533 }
